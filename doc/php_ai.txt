所以我想知道 PHP 該怎麼做到這需求 ?

同時，將帶入 Phalcon 專案中路徑 app/service

-------------------------------------------------

/**
 * 驗證類別是否能成功載入且實例化
 * 留意!! 類別屬性賦值時如果有使用到未存在的值也會報錯，但這部份不用在意
 * @param string $dir_path 資料夾路徑
 */
function runPreloadValidation($dir_path) {
    // 確保路徑結尾沒有斜線，避免 glob 拼接錯誤
    $dir_path = rtrim($dir_path, '/');
    $files = glob($dir_path . '/*.php');

    if (!$files) {
        echo "指定路徑下無 PHP 檔案：{$dir_path} <br>";
        return;
    }

    foreach ($files as $file) {
        $class_name = basename($file, '.php');

        // 1. 檢查類別是否存在 (這會觸發 Phalcon Autoloader)
        if (!class_exists($class_name)) {
            echo "<strong>警告：</strong> 檔案存在但類別名稱 【{$class_name}】 無法載入 (請檢查 Autoloader 設定). <br>";
            continue;
        }

        try {
            // 2. 使用反射機制檢查類別定義
            $reflector = new ReflectionClass($class_name);

            // 檢查是否為抽象類別或介面 (這類不能直接實例化，即使是 newInstanceWithoutConstructor)
            if ($reflector->isAbstract() || $reflector->isInterface()) {
                echo "類別名稱: {$class_name} (Abstract/Interface) 已略過實例化測試. <br>";
                continue;
            }

            // 實例化測試 (不執行建構子)
            $instance = $reflector->newInstanceWithoutConstructor();

            echo "類別名稱: {$class_name} 成功載入並測試實例化. <br>";

            // 釋放資源
            unset($instance, $reflector);
        } catch (ReflectionException $e) {
            echo "反射錯誤 ({$class_name}): " . $e->getMessage() . "<br>";
        } catch (Throwable $t) {
            // PHP 7+ 建議捕捉 Throwable，包含 Error 與 Exception
            echo "系統錯誤 ({$class_name}): " . $t->getMessage() . " 於第 " . $t->getLine() . " 行<br>";
        }
    }
}

// 在 Phalcon 專案中呼叫 (假設路徑正確)
runPreloadValidation(APP_PATH . '/services');
runPreloadValidation(APP_PATH . '/controller');
